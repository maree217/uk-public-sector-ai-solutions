## Automation Scripts

### Script 1: Batch Repository Creator

```bash
#!/bin/bash
# File: create-solution.sh

SOLUTION_NAME=$1
DISPLAY_NAME=$2
DESCRIPTION=$3
TOPICS=$4

if [ -z "$SOLUTION_NAME" ]; then
  echo "Usage: ./create-solution.sh <repo-name> <display-name> <description> <topics>"
  exit 1
fi

echo "Creating solution: $SOLUTION_NAME"

# Create directory
mkdir -p "$SOLUTION_NAME"
cd "$SOLUTION_NAME"

# Copy template files
cp ../SOLUTION-TEMPLATE.md README.md
cp ../citizen-service-assistant/.gitignore .
cp ../citizen-service-assistant/LICENSE .

# Create directory structure
mkdir -p docs/{architecture,integration,deployment}
mkdir -p terraform/{azure-uk,aws-london,on-premise}
mkdir -p src/{api,models,config}
mkdir -p examples/{azure,aws,hybrid}
mkdir -p tests/{unit,integration}

# Replace template placeholders
sed -i '' "s/\[Solution Name\]/$DISPLAY_NAME/g" README.md

# Git setup
git init
git add .
git commit -m "Initial commit: $DISPLAY_NAME"

# Create GitHub repo
gh repo create "$SOLUTION_NAME" --public --source=. \
  --description="$DESCRIPTION" --push

# Add topics
IFS=',' read -ra TOPIC_ARRAY <<< "$TOPICS"
TOPIC_FLAGS=""
for topic in "${TOPIC_ARRAY[@]}"; do
  TOPIC_FLAGS="$TOPIC_FLAGS --add-topic $(echo $topic | xargs)"
done

gh repo edit "maree217/$SOLUTION_NAME" \
  --add-topic uk-public-sector \
  --add-topic production-ready \
  $TOPIC_FLAGS

echo "‚úÖ Created: https://github.com/maree217/$SOLUTION_NAME"
cd ..
```

**Usage:**
```bash
./create-solution.sh \
  grant-writer-ai \
  "Grant Application Assistant" \
  "AI assistant for writing charity and nonprofit grant applications" \
  "third-sector,grant-writing,ai-assistant"
```

### Script 3: Documentation Quality Checker

```bash
#!/bin/bash
# File: check-quality.sh

REPO_NAME=$1

if [ -z "$REPO_NAME" ]; then
  echo "Usage: ./check-quality.sh <repo-name>"
  exit 1
fi

cd "$REPO_NAME"

echo "Checking $REPO_NAME..."

# Check word count (should be 8000+)
WORD_COUNT=$(wc -w < README.md)
echo "Word count: $WORD_COUNT"
if [ "$WORD_COUNT" -lt 8000 ]; then
  echo "‚ö†Ô∏è  README too short (< 8000 words)"
fi

# Check for required sections
REQUIRED_SECTIONS=(
  "## Problem Statement"
  "## Solution"
  "## Integration"
  "## Technical Architecture"
  "## Deployment"
  "## Compliance"
  "## Case Studies"
)

for section in "${REQUIRED_SECTIONS[@]}"; do
  if grep -q "$section" README.md; then
    echo "‚úÖ Found: $section"
  else
    echo "‚ùå Missing: $section"
  fi
done

# Check for UK systems integration
UK_SYSTEMS=(
  "Civica"
  "NEC"
  "Microsoft"
)

echo ""
echo "Integration mentions:"
for system in "${UK_SYSTEMS[@]}"; do
  COUNT=$(grep -c "$system" README.md)
  echo "$system: $COUNT mentions"
done

cd ..
```
